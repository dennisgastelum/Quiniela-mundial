📌 Reglas Principales

• Los pronósticos se cierran antes del pitazo inicial — no se puede editar después.
• Los puntos de clasificación se calculan automáticamente a partir de tus marcadores de fase de grupos: puntos → diferencia de goles → goles marcados.
• En eliminatorias: predice el equipo que avanza y si el partido irá a prórroga/penaltis. El bono de método se otorga independientemente del ganador.
• Desempate general: más resultados correctos → más marcadores exactos → envío más temprano.
• Premio: 1.º se lleva el 50%, 2.º el 30%, 3.º el 20% 🏆

💡 Ejemplo

Pronosticas Brasil 2–1 Marruecos y así termina exactamente:
+2 (resultado) +3 (marcador exacto) = 5 pts.

Si además tu pronóstico implica Brasil 1.º y Marruecos 2.º en el Grupo C:
+4 (Brasil clasifica) +2 (Brasil 1.º) +4 (Marruecos clasifica) = 10 pts más, automáticamente.

var RECAP_EMAILS = ["[email protected]","[email protected]"];

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  if (e.parameter.action === 'getPicks') {
    var s = ss.getSheetByName('Picks') || ss.insertSheet('Picks');
    var d = s.getDataRange().getValues(), p = {};
    for (var i=1;i<d.length;i++) if(d[i][0]) p[d[i][0]]=JSON.parse(d[i][1]||'{}');
    return ContentService.createTextOutput(JSON.stringify({ok:true,picks:p})).setMimeType(ContentService.MimeType.JSON);
  }
  if (e.parameter.action === 'getResults') {
    var s = ss.getSheetByName('Results') || ss.insertSheet('Results');
    var d = s.getDataRange().getValues();
    var r = (d[1]&&d[1][0]) ? JSON.parse(d[1][0]) : {};
    return ContentService.createTextOutput(JSON.stringify({ok:true,results:r})).setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput(JSON.stringify({ok:false})).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var payload = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  if (payload.action === 'savePick') {
    var s = ss.getSheetByName('Picks') || ss.insertSheet('Picks');
    var d = s.getDataRange().getValues(), found = false;
    for (var i=1;i<d.length;i++) {
      if (d[i][0]===payload.name) { s.getRange(i+1,2).setValue(JSON.stringify(payload.picks)); found=true; break; }
    }
    if (!found) { if(!d.length||d[0][0]!=='Name') s.getRange(1,1,1,2).setValues([['Name','Picks']]); s.appendRow([payload.name,JSON.stringify(payload.picks)]); }
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }
  if (payload.action === 'saveResults') {
    var s = ss.getSheetByName('Results') || ss.insertSheet('Results');
    s.clearContents(); s.getRange(1,1,1,2).setValues([['Results','Updated']]);
    s.appendRow([JSON.stringify(payload.results), new Date().toISOString()]);
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput(JSON.stringify({ok:false})).setMimeType(ContentService.MimeType.JSON);
}

function sendDailyRecap() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var ps = ss.getSheetByName('Picks'); if(!ps) return;
  var pd = ps.getDataRange().getValues(), all = {};
  for(var i=1;i<pd.length;i++) if(pd[i][0]) all[pd[i][0]]=JSON.parse(pd[i][1]||'{}');
  if(!Object.keys(all).length) return;
  var rs = ss.getSheetByName('Results'), results = {};
  if(rs){var rd=rs.getDataRange().getValues();if(rd[1]&&rd[1][0])results=JSON.parse(rd[1][0]);}
  var scored=[];
  for(var n in all){var t=simpleScore(all[n],results);scored.push({name:n,total:t});}
  scored.sort(function(a,b){return b.total-a.total;});
  var dateStr=Utilities.formatDate(new Date(),Session.getScriptTimeZone(),'dd MMM yyyy');
  var medals=['🥇','🥈','🥉'], lbHtml='';
  scored.forEach(function(p,i){lbHtml+='<tr><td style="padding:10px 14px;border-bottom:1px solid #1e2e22;">'+(medals[i]||(i+1)+'.')+'</td><td style="padding:10px 14px;border-bottom:1px solid #1e2e22;font-weight:600;">'+p.name+'</td><td style="padding:10px 14px;border-bottom:1px solid #1e2e22;text-align:right;color:#00c853;font-weight:700;">'+p.total+' pts</td></tr>';});
  var leader=scored[0]?scored[0].name:'—', gap=scored.length>1?(scored[0].total-scored[1].total):0;
  var gapTxt=gap>0?leader+' lidera por '+gap+' pts':'¡Empate en la cima!';
  var html='<!DOCTYPE html><html><body style="background:#0a0f0d;font-family:Arial;color:#e8f5e9;"><div style="max-width:600px;margin:0 auto;padding:24px;"><div style="background:linear-gradient(135deg,#0a1f0d,#0d2b10);border:2px solid #00c853;border-radius:12px;padding:24px;text-align:center;margin-bottom:20px;"><h1 style="color:#ffd600;">⚽ QUINIELA MUNDIAL 2026</h1><p style="color:#5a7a60;">RESUMEN — '+dateStr+'</p></div><p style="color:#a5d6a7;margin-bottom:16px;">📊 Líder: <strong style=color:#ffd600>'+leader+'</strong> | '+gapTxt+'</p><div style="background:#111a14;border:1px solid #1e2e22;border-radius:10px;overflow:hidden;"><div style="padding:14px;border-bottom:1px solid #1e2e22;"><h2 style="color:#ffd600;margin:0;">🏆 CLASIFICACIÓN</h2></div><table style="width:100%;border-collapse:collapse;">'+lbHtml+'</table></div></div></body></html>';
  RECAP_EMAILS.forEach(function(em){if(em&&em.indexOf('@')>-1)MailApp.sendEmail({to:em,subject:'⚽ Resumen Quiniela — '+dateStr,htmlBody:html});});
}

function simpleScore(picks, results) {
  var total=0, gids=["A1","A2","A3","A4","A5","A6","B1","B2","B3","B4","B5","B6","C1","C2","C3","C4","C5","C6","D1","D2","D3","D4","D5","D6","E1","E2","E3","E4","E5","E6","F1","F2","F3","F4","F5","F6","G1","G2","G3","G4","G5","G6","H1","H2","H3","H4","H5","H6","I1","I2","I3","I4","I5","I6","J1","J2","J3","J4","J5","J6","K1","K2","K3","K4","K5","K6","L1","L2","L3","L4","L5","L6"];
  var kpts={R32a:5,R32b:5,R32c:5,R32d:5,R32e:5,R32f:5,R32g:5,R32h:5,QFa:8,QFb:8,QFc:8,QFd:8,SF1:12,SF2:12,FIN:20};
  gids.forEach(function(id){var r=results[id],p=picks.matches&&picks.matches[id];if(!r||!r.winner||!p||!p.winner)return;if(p.winner===r.winner){total+=2;if(p.home!==''&&p.away!==''&&String(p.home)===String(r.home)&&String(p.away)===String(r.away))total+=3;}});
  for(var id in kpts){var r=results[id],p=picks.matches&&picks.matches[id];if(r&&r.winner&&p&&p.winner===r.winner)total+=kpts[id];}
  return total;
}
